-----Sightly Demo-----

----------------------Scheme--------------------------
${'//firstdemo.com/path/page.html' @ scheme='http'}
${'http://firstdemo.com/path/page.html' @ scheme='https'}
${'https://firstdemo.com/path/page.html' @ scheme='http'}
${'https://firstdemo.com/path/page.html' @ scheme=''}
${'https://firstdemo.com/path/page.html' @ scheme}
----------------------Domain--------------------------
${'///path/page.html' @ domain='firstdemo.com'}
${'///path/page.html' @ scheme='https', domain='firstdemo.com'}
${'http://firstdemo.com/path/page.html' @ domain='aemdemo.com'}
----------------------Selectors--------------------------
${'///path/page.html' @ selectors='selector1.selector2'}
${'///path/page.html' @ selectors=['selector3', 'selector4']}
${'///path/page.selector3.selector4.html' @ addSelectors=['selector5', 'selector6']}
${'///path/page.selector3.selector4.html' @ removeSelectors=['selector3']}
${'///path/page.selector3.selector4.html' @ selectors=''}
${'///path/page.selector3.selector4.html' @ selectors}
----------------------Extension--------------------------
${'https://firstdemo.com/path/page.html' @ extension='json'}
${'https://firstdemo.com/path/page' @ extension='html'}
${'https://firstdemo.com/path/page.html' @ extension}
----------------------Suffix--------------------------
${'///path/page.html' @ suffix='demo/suffix'}
${'///path/page.htmldemo/suffix' @ prependSuffix='prepended'}
${'///path/page.htmldemo/suffix' @ appendSuffix='appended'}
${'///path/page.htmldemo/suffix' @ suffix}
----------------------Fragment--------------------------
${'https://firstdemo.com/path/page.html' @ fragment='fragment1'}
${'https://firstdemo.com/path/page#fragment1' @ fragment='fragment2'}
${'https://firstdemo.com/path/page.html#fragment2' @ fragment}
